    %utl_submit_r64('                                                                            
    library(haven);                                                                              
    library(data.table);                                                                         
    library(SASxport);                                                                           
    have<-read_sas("d:/sd1/have.sas7bdat");                                                      
    want<-as.data.table(rbind(have,1*(have[1,]==have[2,])));                                     
    write.xport(want,file="d:/xpt/want.xpt");                                                    
    ');                                                                                          
                                                                                                 
    libname xpt xport "d:/xpt/want.xpt";                                                         
    data want;                                                                                   
      set xpt.want;                                                                              
    run;quit;                                                                                    
    libname xpt clear;
